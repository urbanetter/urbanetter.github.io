Key to frontend performance

is to understand the page loading process

In real world, there are a lot more files

This means

  • It took almost 1 second to fetch the initial HTML
  • It took almost 3.5 seconds to reach DOM ready
  • It took almost 4.5 seconds to reach "loaded" event

function render_include_statement(groupName) {
	files = new Group(groupName);
	lastRun = getLastRun();
	newestModificationDate = files.newestModificationDate;
	if (newestModificationDate > lastRun.timestamp) {
		result = concatenate(files);
		result = pack(result);
		filename = md5(newestModificationDate);
		save_file(filename, result);
	} else {
		filename = lastRun.filename;
	}
	return '<script src="filename"></script>';
}
